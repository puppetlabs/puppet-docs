def _assignable?(o)
  return true if self == o
  svalues = values
  if svalues.empty?
    return true if o.is_a?(PStringType) || o.is_a?(PEnumType) || o.is_a?(PPatternType)
  end
  case o
    when PStringType
      # if the set of strings are all found in the set of enums
      !o.values.empty? && o.values.all? { |s| svalues.any? { |e| e == s }}
    when PEnumType
      !o.values.empty? && o.values.all? { |s| svalues.any? {|e| e == s }}
    else
      false
  end
end
