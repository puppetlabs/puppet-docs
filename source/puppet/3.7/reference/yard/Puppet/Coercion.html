def self.boolean(value)
  # downcase strings
  if value.respond_to? :downcase
    value = value.downcase
  end

  case value
  when true, :true, 'true', :yes, 'yes'
    true
  when false, :false, 'false', :no, 'no'
    false
  else
    fail('expected a boolean value')
  end
end
