def adjacent?(o)
  o.is_a?(PIntegerType) &&  (@to + 1 == o.from || o.to + 1 == @from)
end

def merge(o)
  if intersect?(o) || adjacent?(o)
    min = @from <= o.numeric_from ? @from : o.numeric_from
    max = @to >= o.numeric_to ? @to : o.numeric_to
    PIntegerType.new(min, max)
  else
    nil
  end
end
